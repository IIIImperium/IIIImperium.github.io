好的,我将把PWN学习中常用的C语言语句写出来,并详细说明其含义和用法:

  1. 基本数据类型声明
    // 整型变量
    int a = 10; // 有符号整数(通常4字节)
    unsigned int b = 20; // 无符号整数
    long c = 100L; // 长整型(8字节在64位系统)
    short d = 5; // 短整型(2字节)

// 字符型
char ch = ‘A’; // 单字符(1字节)
unsigned char byte = 0xFF; // 常用于表示字节

// 查看类型大小
sizeof(int); // 返回int类型的字节数

  1. 指针操作
    // 指针声明和赋值
    int value = 42;
    int *ptr = &value; // ptr指向value的地址
    *ptr = 100; // 通过指针修改value的值

// 指针运算
int arr[5] = {1,2,3,4,5};
int *p = arr; // 数组名就是指针
p++; // 指针移动到下一个元素(移动sizeof(int)字节)
*(p+2); // 访问arr[3]

// 多级指针
char **argv; // 指向字符指针的指针(常用于命令行参数)
void *generic_ptr; // 通用指针,可转换为任意类型

// 函数指针
int (*func_ptr)(int, int); // 声明函数指针
func_ptr = &some_function; // 赋值
int result = func_ptr(1, 2); // 调用

  1. 危险的字符串函数
    // 【危险】无边界检查的函数
    char buffer[10];

gets(buffer); // 从stdin读取,不检查长度 → 缓冲区溢出!
scanf(“%s”, buffer); // 同样不检查长度 → 危险!
strcpy(buffer, long_string); // 不检查目标缓冲区大小 → 溢出!
strcat(buffer, str); // 字符串拼接,不检查长度 → 危险!
sprintf(buffer, “%s”, str); // 格式化输出,不检查长度 → 危险!

// 【安全】有边界检查的替代函数
fgets(buffer, sizeof(buffer), stdin); // 指定最大读取长度
strncpy(buffer, src, sizeof(buffer)-1); // 限制复制长度
snprintf(buffer, sizeof(buffer), “%s”, str); // 限制输出长度

  1. 格式化字符串相关
    // 【危险】格式化字符串漏洞
    char user_input[100];
    fgets(user_input, 100, stdin);
    printf(user_input); // 危险!用户可控制格式化字符串
    // 如输入”%x %x %x”会泄露栈内容

// 【正确】做法
printf(“%s”, user_input); // 安全:用户输入作为参数,不是格式串

// 格式化说明符
printf(“%d”, num); // 十进制整数
printf(“%x”, num); // 十六进制
printf(“%p”, ptr); // 指针地址
printf(“%s”, str); // 字符串
printf(“%n”, &count); // 将已输出字符数写入count(危险!)

  1. 内存分配函数
    // 动态分配内存
    int arr = (int)malloc(10 * sizeof(int)); // 分配40字节
    if(arr == NULL) { // 检查分配是否成功
    // 处理错误
    }

char str = (char)calloc(100, sizeof(char)); // 分配并清零

arr = (int*)realloc(arr, 20 * sizeof(int)); // 重新调整大小

free(arr); // 释放内存
arr = NULL; // 防止悬空指针

// 【危险】常见漏洞
free(ptr);
free(ptr); // Double Free漏洞!
*ptr = 10; // Use After Free漏洞!

  1. 数组和内存操作
    // 数组声明
    int arr[10]; // 栈上分配
    int matrix[3][4]; // 二维数组

// 数组访问
arr[0] = 100; // 正常访问
arr[15] = 200; // 越界访问(未定义行为,可能崩溃或被利用)

// 内存操作函数
memset(buffer, 0, sizeof(buffer)); // 将内存设置为指定值
memcpy(dest, src, n); // 复制n字节内存
memmove(dest, src, n); // 安全的内存复制(处理重叠)
memcmp(buf1, buf2, n); // 比较内存内容

  1. 结构体
    // 定义结构体
    struct Person {
    char name[50];
    int age;
    char *address; // 指针成员
    };

// 使用结构体
struct Person p1;
p1.age = 25; // 访问成员
strcpy(p1.name, “Alice”);

struct Person *ptr = &p1;
ptr->age = 30; // 通过指针访问成员

// 结构体内存布局(考虑对齐)
struct Data {
char a; // 1字节
// 填充3字节
int b; // 4字节
char c; // 1字节
// 填充3字节
}; // 总共12字节(不是6字节)

  1. 系统调用和危险函数
    // 执行系统命令
    system(“/bin/sh”); // 获取shell(PWN的常见目标)
    system(user_input); // 【危险】命令注入漏洞!

// 进程相关
execve(“/bin/sh”, argv, envp); // 执行程序
fork(); // 创建子进程

// 文件操作
int fd = open(“file.txt”, O_RDONLY);
read(fd, buffer, size);
write(fd, data, size);
close(fd);

  1. 位操作
    // 位运算符
    int a = 0x12;
    int b = 0x34;

a & b; // 按位与
a | b; // 按位或
a ^ b; // 按位异或
~a; // 按位取反
a << 2; // 左移2位(相当于乘4)
a >> 2; // 右移2位(相当于除4)

// 位掩码操作
int flags = 0;
flags |= 0x01; // 设置第0位
flags &= ~0x02; // 清除第1位
if(flags & 0x04) {} // 检查第2位

  1. 条件和循环
    // 条件语句
    if(ptr == NULL) {
    // 空指针检查
    } else if(size > MAX_SIZE) {
    // 大小检查
    }

// 三元运算符
int max = (a > b) ? a : b;

// 循环
for(int i = 0; i < 10; i++) {
// 循环体
}

while(condition) {
// 循环体
}

do {
// 至少执行一次
} while(condition);

  1. 类型转换(重要)
    // 隐式转换可能导致问题
    unsigned int a = -1; // a变成4294967295(整数溢出)
    int size = user_input;
    if(size > 0) {
    malloc(size); // 如果user_input很大,可能溢出
    }

// 显式转换
int x = 1000;
char c = (char)x; // 截断为-24(只保留低8位)

// 指针类型转换
void *ptr = malloc(100);
int int_ptr = (int)ptr; // 类型转换

  1. 内联汇编(高级)
    // 在C代码中嵌入汇编
    asm(“nop”); // 插入空操作指令

// 获取栈指针
register long rsp asm(“rsp”);
printf(“Stack pointer: %p\n”, (void*)rsp);

实际PWN示例代码
// 一个包含栈溢出漏洞的简单程序
#include <stdio.h>
#include <string.h>

void vulnerable_function() {
char buffer[64]; // 64字节缓冲区
printf(“Enter your name: “);
gets(buffer); // 漏洞点!没有长度检查
printf(“Hello, %s!\n”, buffer);
}

int main() {
vulnerable_function();
return 0;
}

// 利用方式:输入超过64字节的数据,覆盖返回地址
